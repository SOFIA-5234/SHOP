  clearCart: () => void;
  getCartTotal: () => number;

  setIsLoggedIn: (login: boolean) => void;

  couponCode: string;
  setCouponCode: (code: string) => void;

  applyCoupon: () => boolean;
  categoryFilter: string;
  setCategoryFilter: (cat: string) => void;
  searchQuery: string;

}

const ShopContext = createContext<ShopContextType | undefined>(undefined);

export function ShopProvider({ children }: { children: ReactNode }) {
  const [currentPage, setCurrentPage] = useState<ViewState>('home');
  const [selectedProduct, setSelectedProduct] = useState<Product | null>(null);
  const [products, setProducts] = useState<Product[]>(INITIAL_PRODUCTS);
  const [cart, setCart] = useState<CartItem[]>([]);
  const [orders, setOrders] = useState<Order[]>([]);
  const [currentOrder, setCurrentOrder] = useState<Order | null>(null);
  const [isLoggedIn, setIsLoggedIn] = useState(false);
  const [userEmail, setUserEmail] = useState('');
  const [couponCode, setCouponCode] = useState('');
  const [activeDiscount, setActiveDiscount] = useState<{ code: string; percent: number } | null>(null);
  const [categoryFilter, setCategoryFilter] = useState('All');
  const [searchQuery, setSearchQuery] = useState('');

  const addToCart = (product: Product, plan = 'รายเดือน') => {
    const existing = cart.find(item => item.product.id === product.id && item.plan === plan);
    if (existing) {
      setCart(cart.map(item => item.product.id === product.id && item.plan === plan ? { ...item, quantity: item.quantity + 1 } : item));
    } else {
      setCart([...cart, { product, quantity: 1, plan }]);
    }
    setCurrentPage('cart');
  };

  const removeFromCart = (productId: string, plan: string) => {
    setCart(cart.filter(item => !(item.product.id === productId && item.plan === plan)));
  };

  const clearCart = () => setCart([]);

  const getCartTotal = () => {
    const total = cart.reduce((sum, item) => sum + (item.product.price * item.quantity), 0);
    if (activeDiscount) {
      return total - (total * (activeDiscount.percent / 100));
    }
    return total;
  };

  const applyCoupon = () => {
    if (couponCode.toUpperCase() === 'SOFIA50') {
      setActiveDiscount({ code: 'SOFIA50', percent: 50 });
      return true;
    }
    return false;
  };

  const createOrder = () => {
    if (cart.length === 0) return;
    const newOrder: Order = {
      id: 'SF-' + Math.floor(100000 + Math.random() * 900000),
      items: [...cart],
      totalPrice: getCartTotal(),
      status: 'pending',
      paymentMethod: 'PromptPay QR',
      slipUploaded: false,
      createdAt: new Date().toLocaleDateString('th-TH', { hour: '2-digit', minute: '2-digit' })
    };
    setOrders([newOrder, ...orders]);
    setCurrentOrder(newOrder);
    clearCart();
    setCurrentPage('checkout');
  };

  const uploadSlip = (orderId: string) => {
    setOrders(prev => prev.map(o => o.id === orderId ? { ...o, status: 'verifying', slipUploaded: true } : o));
    if (currentOrder && currentOrder.id === orderId) {
      setCurrentOrder({ ...currentOrder, status: 'verifying', slipUploaded: true });
    }
    setCurrentPage('success');
  };

  return (
    <ShopContext.Provider value={{
      currentPage, setCurrentPage,
      selectedProduct, setSelectedProduct,
      products, setProducts,
      cart, addToCart, removeFromCart, clearCart, getCartTotal,
      orders, setOrders,
      currentOrder, setCurrentOrder,
      createOrder, uploadSlip,
      isLoggedIn, setIsLoggedIn,
      userEmail, setUserEmail,
      couponCode, setCouponCode,
      activeDiscount, applyCoupon,
      categoryFilter, setCategoryFilter,
      searchQuery, setSearchQuery
    }}>
      {children}
    </ShopContext.Provider>
  );
}

export function useShop() {
  const context = useContext(ShopContext);
  if (!context) throw new Error('useShop must be used within a ShopProvider');
  return context;
}
