'use client';

import React from 'react';
import { AnimatePresence } from 'framer-motion';
import { useShop } from '../context/ShopContext';

// Components
import Navbar from '../components/Navbar';
import Footer from '../components/Footer';
import FloatingSupport from '../components/FloatingSupport';

// Views
import HomeView from '../components/views/HomeView';
import ShopView from '../components/views/ShopView';
import DetailView from '../components/views/DetailView';
import LoginView from '../components/views/LoginView';
import CartView from '../components/views/CartView';
import CheckoutView from '../components/views/CheckoutView';
import SuccessView from '../components/views/SuccessView';
import TrackingView from '../components/views/TrackingView';
import ContactView from '../components/views/ContactView';
import FAQView from '../components/views/FAQView';

export default function RootPage() {
  const { currentPage } = useShop();

  return (
    <div className="bg-[#030014] text-slate-100 min-h-screen relative flex flex-col justify-between overflow-x-hidden">
      
      <div className="absolute top-0 left-1/4 w-[500px] h-[500px] bg-purple-900/10 rounded-full blur-[120px] pointer-events-none animate-pulse-slow"></div>
      <div className="absolute top-1/3 right-1/4 w-[600px] h-[600px] bg-pink-900/10 rounded-full blur-[150px] pointer-events-none"></div>

      <div>
        <Navbar />
        <main className="max-w-7xl mx-auto px-4 md:px-8 py-8">
          <AnimatePresence mode="wait">
            {currentPage === 'home' && <HomeView />}
            {currentPage === 'shop' && <ShopView />}
            {currentPage === 'detail' && <DetailView />}
            {currentPage === 'login' && <LoginView />}
            {currentPage === 'cart' && <CartView />}
            {currentPage === 'checkout' && <CheckoutView />}
            {currentPage === 'success' && <SuccessView />}
            {currentPage === 'tracking' && <TrackingView />}
            {currentPage === 'contact' && <ContactView />}
            {currentPage === 'faq' && <FAQView />}
          </AnimatePresence>
        </main>
      </div>

      <Footer />
      <FloatingSupport />
    </div>
  );
}
